SELECT pg_terminate_backend(pg_stat_activity.pid)
FROM pg_stat_activity
WHERE pg_stat_activity.datname = 'airbox-locationmanagement'
AND pid <> pg_backend_pid();

DROP DATABASE "airbox-locationmanagement";

DROP ROLE liquibase_airbox_locationmanagement;

DROP ROLE airboxlmsuser;
